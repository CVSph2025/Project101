# Social Authentication Configuration
# 
# To enable Facebook and Google social authentication, you need to:
# 1. Create applications in Facebook and Google developer consoles
# 2. Add the following environment variables to your .env file
# 3. Configure the redirect URLs in your applications

# Facebook OAuth Configuration
# Create app at: https://developers.facebook.com/
FACEBOOK_CLIENT_ID=your_facebook_app_id_here
FACEBOOK_CLIENT_SECRET=your_facebook_app_secret_here
FACEBOOK_REDIRECT_URL="${APP_URL}/auth/facebook/callback"

# Google OAuth Configuration  
# Create app at: https://console.developers.google.com/
GOOGLE_CLIENT_ID=your_google_client_id_here
GOOGLE_CLIENT_SECRET=your_google_client_secret_here
GOOGLE_REDIRECT_URL="${APP_URL}/auth/google/callback"

# Steps to configure:

# Facebook App Setup:
# 1. Go to https://developers.facebook.com/
# 2. Create a new app, select "Consumer" type
# 3. Add "Facebook Login" product
# 4. In Facebook Login settings, add redirect URL: http://localhost:8000/auth/facebook/callback
# 5. For production, add: https://yourdomain.com/auth/facebook/callback
# 6. Copy App ID and App Secret to .env file

# Google App Setup:
# 1. Go to https://console.developers.google.com/
# 2. Create a new project or select existing
# 3. Enable Google+ API and Google OAuth2 API
# 4. Create OAuth 2.0 credentials
# 5. Add authorized redirect URI: http://localhost:8000/auth/google/callback
# 6. For production, add: https://yourdomain.com/auth/google/callback
# 7. Copy Client ID and Client Secret to .env file

# Testing URLs:
# Facebook Login: http://localhost:8000/auth/facebook
# Google Login: http://localhost:8000/auth/google
